flume安装

连接kafka命令

连接启动 /root/soft/flume1.8/bin/flume-ng agent -c conf/ -n a1 -f /root/soft/flume1.8/conf/flume-2-kafka.conf

项目中使用的
flume-2-kafka.conf文件 新建一个

#将kafka节点改成自己的
#把文件放到flume1.8/conf/flume-2-kafka.conf下
a1.sources = r1
a1.channels = c1
a1.sinks = k1

a1.sources.r1.type = exec
a1.sources.r1.command = tail -F -c +0 /root/soft/ct/call.log

a1.channels.c1.type = memory
a1.channels.c1.capacity = 1000
a1.channels.c1.transactionCapacity = 100
a1.channels.c1.byteCapacityBufferPercentage = 20
a1.channels.c1.byteCapacity = 800000

a1.sinks.k1.type = org.apache.flume.sink.kafka.KafkaSink
a1.sinks.k1.kafka.topic = ct
a1.sinks.k1.kafka.bootstrap.servers = node1:9092,node2:9092,node3:9092
a1.sinks.k1.kafka.flumeBatchSize = 20
a1.sinks.k1.kafka.producer.acks = 1

a1.sources.r1.channels = c1
a1.sinks.k1.channel = c1
