import numpy as np
arr = [1,2,4,1,7,8,3]
def rec_opt(arr,i):
    if i == 0:
        return arr[0]
    elif i == 1:
        return max(arr[0],arr[1])
    else:
        A = rec_opt(arr, i - 2) + arr[i]
        B = rec_opt(arr, i - 1)
        return max(A,B)

#递归会产生重叠子问题 复杂度N方

def dp_opt(arr):
    opt = np.zeros(len(arr))
    opt[0] = arr[0]
    opt[1] = max(arr[0],arr[1])
    for i in range(2,len(arr)):
        A = opt[i - 2] + arr[i]
        B = opt[i - 1]
        opt[i] = max(A,B)
    return opt[len(arr) - 1]

a = dp_opt(arr)
print(a)

arr1 = [3,34,4,12,5,2]
#递归的方式求
def rec_subset(arr,i,s):
    #三种递归出口
    if(s == 0):
        return True
    elif i == 0:
        return arr[0] == s
    elif arr[i] > s:
        return rec_subset(arr,i-1,s)
    else:
        A = rec_subset(arr, i-1,s-arr[i])#选这个
        B = rec_subset(arr,i-1,s)#不选这个数
        return A or B
s = rec_subset(arr1,len(arr1)-1,13)
print(s)

def dp_subset(arr,s):
    subset = np.zeros((len(arr),s+1),dtype=bool)
    subset[:,0] = True
    subset[0,:] = False
    subset[0,arr[0]] = True
    #填表
    for i in range(1,len(arr)):
        for s in range(1,s+1):
            if(arr[i] > s):
                subset[i,s] = subset[i-1,s]
            else:
                A = subset[i-1,s-arr[i]]
                B = subset[i-1,s]
                subset[i,s] = A or B
    r,c = subset.shape
    return subset[r-1,c-1]
v = dp_subset(arr1,9)
print(v)
